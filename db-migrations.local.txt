CREATE EXTENSION IF NOT EXISTS "pgcrypto";

-- ---------------------------------------------------------------
-- tenants
-- ---------------------------------------------------------------
CREATE TABLE "tenants" (
    "id"              UUID NOT NULL DEFAULT gen_random_uuid(),
    "slug"            VARCHAR(63) NOT NULL,
    "name"            VARCHAR(120) NOT NULL,
    "logo_url"        TEXT,
    "color_primary"   VARCHAR(7) NOT NULL DEFAULT 'blue',
    "color_secondary" VARCHAR(7) NOT NULL DEFAULT 'purple',
    "status"          VARCHAR(20) NOT NULL DEFAULT 'active',
    "feature_flags"   JSONB NOT NULL DEFAULT '{}',
    "custom_texts"    JSONB NOT NULL DEFAULT '{}',
    "created_at"      TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updated_at"      TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT "tenants_pkey" PRIMARY KEY ("id")
);

CREATE UNIQUE INDEX "tenants_slug_key" ON "tenants"("slug");

-- ---------------------------------------------------------------
-- users
-- ---------------------------------------------------------------
CREATE TABLE "users" (
    "id"                        UUID NOT NULL DEFAULT gen_random_uuid(),
    "name"                      TEXT NOT NULL,
    "email"                     TEXT NOT NULL,
    "password_hash"             TEXT,
    "phone"                     TEXT,
    "avatar_url"                TEXT,
    "role"                      TEXT NOT NULL DEFAULT 'agent',
    "created_at"                TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updated_at"                TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "password_reset_token_hash" TEXT,
    "password_reset_expires_at" TIMESTAMPTZ(6),
    "google_id"                 TEXT,
    "tenant_id"                 UUID,
    "status"                    VARCHAR(20) NOT NULL DEFAULT 'active',
    CONSTRAINT "users_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "users_tenant_id_fkey"
        FOREIGN KEY ("tenant_id") REFERENCES "tenants"("id")
        ON DELETE NO ACTION ON UPDATE NO ACTION
);

CREATE UNIQUE INDEX "users_email_key"                     ON "users"("email");
CREATE UNIQUE INDEX "users_password_reset_token_hash_key" ON "users"("password_reset_token_hash");
CREATE UNIQUE INDEX "users_google_id_key"                 ON "users"("google_id");
CREATE INDEX        "idx_users_tenant_id"                 ON "users"("tenant_id");

-- ---------------------------------------------------------------
-- user_settings
-- ---------------------------------------------------------------
CREATE TABLE "user_settings" (
    "id"                         UUID NOT NULL DEFAULT gen_random_uuid(),
    "user_id"                    UUID NOT NULL,
    "whatsapp_phone_id"          TEXT,
    "whatsapp_token"             TEXT,
    "whatsapp_webhook_token"     TEXT,
    "ai_provider"                TEXT NOT NULL DEFAULT 'openai',
    "ai_api_key"                 TEXT,
    "ai_model"                   TEXT NOT NULL DEFAULT 'gpt-4o-mini',
    "greeting_message"           TEXT,
    "auto_reply_enabled"         BOOLEAN NOT NULL DEFAULT true,
    "follow_up_enabled"          BOOLEAN NOT NULL DEFAULT true,
    "follow_up_delay_hours"      INTEGER NOT NULL DEFAULT 24,
    "max_follow_ups"             INTEGER NOT NULL DEFAULT 3,
    "auto_reply_delay_seconds"   INTEGER NOT NULL DEFAULT 0,
    "facebook_page_access_token" TEXT,
    "facebook_auto_outreach"     BOOLEAN NOT NULL DEFAULT true,
    "openai_transcription_key"   TEXT,
    "elevenlabs_voice_id"        TEXT,
    "voice_reply_enabled"        BOOLEAN NOT NULL DEFAULT false,
    "voice_reply_monthly_limit"  INTEGER NOT NULL DEFAULT 50,
    CONSTRAINT "user_settings_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "user_settings_user_id_fkey"
        FOREIGN KEY ("user_id") REFERENCES "users"("id")
        ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE UNIQUE INDEX "user_settings_user_id_key" ON "user_settings"("user_id");

-- ---------------------------------------------------------------
-- facebook_page_mappings
-- ---------------------------------------------------------------
CREATE TABLE "facebook_page_mappings" (
    "id"         UUID NOT NULL DEFAULT gen_random_uuid(),
    "page_id"    TEXT NOT NULL,
    "user_id"    UUID NOT NULL,
    "page_name"  TEXT,
    "created_at" TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
    CONSTRAINT "facebook_page_mappings_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "facebook_page_mappings_user_id_fkey"
        FOREIGN KEY ("user_id") REFERENCES "users"("id")
        ON DELETE CASCADE ON UPDATE NO ACTION
);

CREATE UNIQUE INDEX "facebook_page_mappings_page_id_key" ON "facebook_page_mappings"("page_id");

-- ---------------------------------------------------------------
-- pipeline_stages
-- ---------------------------------------------------------------
CREATE TABLE "pipeline_stages" (
    "id"         UUID NOT NULL DEFAULT gen_random_uuid(),
    "user_id"    UUID NOT NULL,
    "name"       TEXT NOT NULL,
    "color"      TEXT NOT NULL DEFAULT 'blue',
    "order"      INTEGER NOT NULL DEFAULT 0,
    "created_at" TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT "pipeline_stages_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "pipeline_stages_user_id_fkey"
        FOREIGN KEY ("user_id") REFERENCES "users"("id")
        ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE UNIQUE INDEX "pipeline_stages_user_id_order_key" ON "pipeline_stages"("user_id", "order");

-- ---------------------------------------------------------------
-- leads
-- ---------------------------------------------------------------
CREATE TABLE "leads" (
    "id"                UUID NOT NULL DEFAULT gen_random_uuid(),
    "user_id"           UUID NOT NULL,
    "name"              TEXT NOT NULL,
    "phone"             TEXT NOT NULL,
    "email"             TEXT,
    "avatar_url"        TEXT,
    "source"            TEXT NOT NULL DEFAULT 'whatsapp',
    "status"            TEXT NOT NULL DEFAULT 'new',
    "score"             INTEGER NOT NULL DEFAULT 0,
    "value"             DECIMAL(12,2),
    "region"            TEXT,
    "price_min"         DECIMAL(12,2),
    "price_max"         DECIMAL(12,2),
    "property_type"     TEXT,
    "purpose"           TEXT,
    "timeline"          TEXT,
    "bedrooms"          INTEGER,
    "notes"             TEXT,
    "pipeline_stage_id" UUID,
    "last_contact_at"   TIMESTAMP(3),
    "next_follow_up_at" TIMESTAMP(3),
    "follow_up_count"   INTEGER NOT NULL DEFAULT 0,
    "created_at"        TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updated_at"        TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT "leads_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "leads_user_id_fkey"
        FOREIGN KEY ("user_id") REFERENCES "users"("id")
        ON DELETE CASCADE ON UPDATE CASCADE,
    CONSTRAINT "leads_pipeline_stage_id_fkey"
        FOREIGN KEY ("pipeline_stage_id") REFERENCES "pipeline_stages"("id")
        ON DELETE SET NULL ON UPDATE CASCADE
);

CREATE INDEX "leads_user_id_status_idx"    ON "leads"("user_id", "status");
CREATE INDEX "leads_user_id_score_idx"     ON "leads"("user_id", "score" DESC);
CREATE INDEX "leads_next_follow_up_at_idx" ON "leads"("next_follow_up_at");

-- ---------------------------------------------------------------
-- conversations
-- ---------------------------------------------------------------
CREATE TABLE "conversations" (
    "id"               UUID NOT NULL DEFAULT gen_random_uuid(),
    "lead_id"          UUID NOT NULL,
    "whatsapp_chat_id" TEXT,
    "status"           TEXT NOT NULL DEFAULT 'active',
    "unread_count"     INTEGER NOT NULL DEFAULT 0,
    "last_message_at"  TIMESTAMP(3),
    "created_at"       TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updated_at"       TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT "conversations_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "conversations_lead_id_fkey"
        FOREIGN KEY ("lead_id") REFERENCES "leads"("id")
        ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE UNIQUE INDEX "conversations_lead_id_key" ON "conversations"("lead_id");

-- ---------------------------------------------------------------
-- messages
-- ---------------------------------------------------------------
CREATE TABLE "messages" (
    "id"              UUID NOT NULL DEFAULT gen_random_uuid(),
    "conversation_id" UUID NOT NULL,
    "direction"       TEXT NOT NULL,
    "sender"          TEXT NOT NULL,
    "content"         TEXT NOT NULL,
    "type"            TEXT NOT NULL DEFAULT 'text',
    "status"          TEXT NOT NULL DEFAULT 'sent',
    "whatsapp_msg_id" TEXT,
    "metadata"        JSONB,
    "created_at"      TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT "messages_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "messages_conversation_id_fkey"
        FOREIGN KEY ("conversation_id") REFERENCES "conversations"("id")
        ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE UNIQUE INDEX "messages_conversation_id_whatsapp_msg_id_key"
    ON "messages"("conversation_id", "whatsapp_msg_id");
CREATE INDEX "messages_conversation_id_created_at_idx"
    ON "messages"("conversation_id", "created_at");

-- ---------------------------------------------------------------
-- activities
-- ---------------------------------------------------------------
CREATE TABLE "activities" (
    "id"          UUID NOT NULL DEFAULT gen_random_uuid(),
    "user_id"     UUID NOT NULL,
    "lead_id"     UUID,
    "type"        TEXT NOT NULL,
    "title"       TEXT NOT NULL,
    "description" TEXT,
    "metadata"    JSONB,
    "created_at"  TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT "activities_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "activities_user_id_fkey"
        FOREIGN KEY ("user_id") REFERENCES "users"("id")
        ON DELETE CASCADE ON UPDATE CASCADE,
    CONSTRAINT "activities_lead_id_fkey"
        FOREIGN KEY ("lead_id") REFERENCES "leads"("id")
        ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE INDEX "activities_lead_id_created_at_idx" ON "activities"("lead_id", "created_at");

-- ---------------------------------------------------------------
-- lead_actions
-- ---------------------------------------------------------------
CREATE TABLE "lead_actions" (
    "id"           UUID NOT NULL DEFAULT gen_random_uuid(),
    "user_id"      UUID NOT NULL,
    "lead_id"      UUID NOT NULL,
    "type"         VARCHAR NOT NULL,
    "status"       VARCHAR NOT NULL DEFAULT 'pending',
    "title"        VARCHAR NOT NULL,
    "notes"        TEXT,
    "origin"       VARCHAR NOT NULL DEFAULT 'manual',
    "scheduled_at" TIMESTAMPTZ(6),
    "reminder_at"  TIMESTAMPTZ(6),
    "completed_at" TIMESTAMPTZ(6),
    "created_at"   TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
    "updated_at"   TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
    CONSTRAINT "lead_actions_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "lead_actions_user_id_fkey"
        FOREIGN KEY ("user_id") REFERENCES "users"("id")
        ON DELETE CASCADE ON UPDATE NO ACTION,
    CONSTRAINT "lead_actions_lead_id_fkey"
        FOREIGN KEY ("lead_id") REFERENCES "leads"("id")
        ON DELETE CASCADE ON UPDATE NO ACTION
);

CREATE INDEX "idx_lead_actions_lead_type_status"
    ON "lead_actions"("lead_id", "type", "status");
CREATE INDEX "idx_lead_actions_user_status_scheduled"
    ON "lead_actions"("user_id", "status", "scheduled_at");

-- ---------------------------------------------------------------
-- appointments
-- ---------------------------------------------------------------
CREATE TABLE "appointments" (
    "id"                   UUID NOT NULL DEFAULT gen_random_uuid(),
    "user_id"              UUID NOT NULL,
    "lead_id"              UUID NOT NULL,
    "lead_action_id"       UUID,
    "title"                VARCHAR(255) NOT NULL,
    "scheduled_at"         TIMESTAMPTZ(6) NOT NULL,
    "duration_minutes"     INTEGER NOT NULL DEFAULT 60,
    "address"              TEXT,
    "notes"                TEXT,
    "status"               VARCHAR(30) NOT NULL DEFAULT 'pending_confirmation',
    "google_event_id"      VARCHAR(255),
    "google_calendar_id"   VARCHAR(255),
    "confirmation_sent_at" TIMESTAMPTZ(6),
    "confirmation_reply"   VARCHAR(30),
    "created_at"           TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
    "updated_at"           TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
    CONSTRAINT "appointments_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "appointments_user_id_fkey"
        FOREIGN KEY ("user_id") REFERENCES "users"("id")
        ON DELETE CASCADE ON UPDATE NO ACTION,
    CONSTRAINT "appointments_lead_id_fkey"
        FOREIGN KEY ("lead_id") REFERENCES "leads"("id")
        ON DELETE CASCADE ON UPDATE NO ACTION,
    CONSTRAINT "appointments_lead_action_id_fkey"
        FOREIGN KEY ("lead_action_id") REFERENCES "lead_actions"("id")
        ON UPDATE NO ACTION
);

CREATE INDEX "idx_appointments_lead_id"        ON "appointments"("lead_id");
CREATE INDEX "idx_appointments_user_scheduled" ON "appointments"("user_id", "scheduled_at");
CREATE INDEX "idx_appointments_user_status"    ON "appointments"("user_id", "status");

-- ---------------------------------------------------------------
-- calendar_connections
-- ---------------------------------------------------------------
CREATE TABLE "calendar_connections" (
    "id"               UUID NOT NULL DEFAULT gen_random_uuid(),
    "user_id"          UUID NOT NULL,
    "provider"         VARCHAR(20) NOT NULL DEFAULT 'google',
    "access_token"     TEXT,
    "refresh_token"    TEXT,
    "token_expires_at" TIMESTAMPTZ(6),
    "calendar_id"      VARCHAR(255) NOT NULL DEFAULT 'primary',
    "created_at"       TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
    "updated_at"       TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
    CONSTRAINT "calendar_connections_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "calendar_connections_user_id_fkey"
        FOREIGN KEY ("user_id") REFERENCES "users"("id")
        ON DELETE CASCADE ON UPDATE NO ACTION
);

CREATE UNIQUE INDEX "calendar_connections_user_id_provider_key"
    ON "calendar_connections"("user_id", "provider");
CREATE INDEX "idx_calendar_connections_user_id" ON "calendar_connections"("user_id");

-- ---------------------------------------------------------------
-- invite_tokens
-- ---------------------------------------------------------------
CREATE TABLE "invite_tokens" (
    "id"         UUID NOT NULL DEFAULT gen_random_uuid(),
    "tenant_id"  UUID NOT NULL,
    "token"      VARCHAR(64) NOT NULL,
    "email"      VARCHAR(255),
    "role"       VARCHAR(20) NOT NULL DEFAULT 'agent',
    "max_uses"   INTEGER NOT NULL DEFAULT 1,
    "used_count" INTEGER NOT NULL DEFAULT 0,
    "expires_at" TIMESTAMPTZ(6) NOT NULL,
    "created_at" TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT "invite_tokens_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "invite_tokens_tenant_id_fkey"
        FOREIGN KEY ("tenant_id") REFERENCES "tenants"("id")
        ON DELETE CASCADE ON UPDATE NO ACTION
);

CREATE UNIQUE INDEX "invite_tokens_token_key" ON "invite_tokens"("token");
CREATE INDEX        "idx_invite_tokens_token"  ON "invite_tokens"("token");

-- ---------------------------------------------------------------
-- voice_reply_usage
-- ---------------------------------------------------------------
CREATE TABLE "voice_reply_usage" (
    "id"         UUID NOT NULL DEFAULT gen_random_uuid(),
    "user_id"    UUID NOT NULL,
    "month"      VARCHAR(7) NOT NULL,
    "count"      INTEGER NOT NULL DEFAULT 0,
    "updated_at" TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
    CONSTRAINT "voice_reply_usage_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "voice_reply_usage_user_id_fkey"
        FOREIGN KEY ("user_id") REFERENCES "users"("id")
        ON DELETE CASCADE ON UPDATE NO ACTION,
    CONSTRAINT "voice_reply_usage_user_id_month_key"
        UNIQUE ("user_id", "month")
);

CREATE INDEX "idx_voice_reply_usage_user_month" ON "voice_reply_usage"("user_id", "month");

-- ---------------------------------------------------------------
-- properties
-- ---------------------------------------------------------------
CREATE TABLE "properties" (
    "id"            UUID NOT NULL DEFAULT gen_random_uuid(),
    "user_id"       UUID NOT NULL,
    "raw_text"      TEXT NOT NULL,
    "title"         TEXT,
    "type"          TEXT,
    "purpose"       TEXT,
    "price"         DECIMAL(12,2),
    "area"          DECIMAL(8,2),
    "bedrooms"      INTEGER,
    "bathrooms"     INTEGER,
    "parking_spots" INTEGER,
    "address"       TEXT,
    "neighborhood"  TEXT,
    "city"          TEXT,
    "state"         TEXT,
    "amenities"     TEXT[] NOT NULL DEFAULT '{}',
    "description"   TEXT,
    "created_at"    TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
    "updated_at"    TIMESTAMPTZ(6) NOT NULL DEFAULT now(),
    "pdf_url"       TEXT,
    "pdf_filename"  VARCHAR(255),
    "pdf_size"      INTEGER,
    "pdfs"          JSONB NOT NULL DEFAULT '[]',
    CONSTRAINT "properties_pkey" PRIMARY KEY ("id"),
    CONSTRAINT "properties_user_id_fkey"

CREATE INDEX "Property_userId_idx" ON "properties"("user_id", "created_at" DESC);
